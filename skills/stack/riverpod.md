# Riverpod 2.x — State Management Patterns

Read this before writing any Flutter state management code.

---

## Provider Type Selection

Choose the right provider for the job:

| Provider | Use when | Example |
|---|---|---|
| `Provider` | Sync, no state, dependency injection | API service instance |
| `FutureProvider` | Async one-time fetch, no mutation | Config, initial data load |
| `StreamProvider` | Realtime data stream | WebSocket feed, Firestore stream |
| `StateProvider` | Simple, single-value mutable state | UI toggle, counter |
| `NotifierProvider` | Complex sync state with methods | Shopping cart, form state |
| `AsyncNotifierProvider` | Complex async state with methods | Auth, paginated list |

**Default**: When in doubt, use `AsyncNotifierProvider`. It handles loading/error/data states automatically.

---

## AsyncNotifierProvider — Most Important Pattern

Used for auth, API-fetched data with mutations.

```dart
// lib/features/habits/providers/habits_provider.dart
import 'package:riverpod_annotation/riverpod_annotation.dart';

part 'habits_provider.g.dart';  // Generated by build_runner

@riverpod
class Habits extends _$Habits {
  @override
  Future<List<Habit>> build() async {
    // Called on first watch and on ref.invalidateSelf()
    return ref.read(habitsApiProvider).getHabits();
  }

  Future<void> createHabit(String name, String frequency) async {
    state = const AsyncValue.loading();
    state = await AsyncValue.guard(() async {
      await ref.read(habitsApiProvider).createHabit(name, frequency);
      return ref.read(habitsApiProvider).getHabits();  // Refresh list
    });
  }

  Future<void> deleteHabit(String id) async {
    state = await AsyncValue.guard(() async {
      await ref.read(habitsApiProvider).deleteHabit(id);
      final current = state.requireValue;
      return current.where((h) => h.id != id).toList();  // Optimistic update
    });
  }
}

// Read-only with auto-dispose
@riverpod
Future<Habit> habitDetail(HabitDetailRef ref, String id) async {
  return ref.read(habitsApiProvider).getHabit(id);
}
```

## API Service Provider (Dependency Injection)

```dart
// lib/core/providers/api_client_provider.dart
import 'package:riverpod_annotation/riverpod_annotation.dart';
import '../network/api_client.dart';

part 'api_client_provider.g.dart';

  return ApiClient(Env.apiBaseUrl);

  return HabitsApi(ref.read(apiClientProvider));
```

## ConsumerWidget vs ConsumerStatefulWidget

```dart
// ConsumerWidget — for stateless widgets that need providers
class HabitsList extends ConsumerWidget {
  const HabitsList({super.key});

  @override
  Widget build(BuildContext context, WidgetRef ref) {
    // ref.watch — reactive, rebuilds when value changes
    final habitsAsync = ref.watch(habitsProvider);
    
    return habitsAsync.when(
      loading: () => const CircularProgressIndicator(),
      error: (err, _) => Text('Error: $err'),
      data: (habits) => ListView.builder(
        itemCount: habits.length,
        itemBuilder: (context, i) => HabitTile(habit: habits[i]),
      ),
    );
  }
}

// ConsumerStatefulWidget — when you need initState, dispose, or local state
class HabitForm extends ConsumerStatefulWidget {
  const HabitForm({super.key});

  @override
  ConsumerState<HabitForm> createState() => _HabitFormState();
}

class _HabitFormState extends ConsumerState<HabitForm> {
  final _nameController = TextEditingController();
  
  @override
  void dispose() {
    _nameController.dispose();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    return TextField(controller: _nameController);
  }
  
  Future<void> _submit() async {
    // ref.read — one-time read in callbacks/methods
    await ref.read(habitsProvider.notifier).createHabit(_nameController.text, 'daily');
  }
}
```

## ref.watch vs ref.read vs ref.listen

This is the most commonly confused concept:

```dart
// ref.watch — ONLY in build() — reactive
// Re-runs build() when value changes
@override
Widget build(BuildContext context, WidgetRef ref) {
  final auth = ref.watch(authProvider);  // ✅ Correct place
  // ...
}

// ref.read — in callbacks, methods, outside build
// Does NOT subscribe — just reads current value once
onPressed: () async {
  await ref.read(authProvider.notifier).login(email, password);  // ✅
}

// ref.listen — react to changes with side effects (outside build)
@override
Widget build(BuildContext context, WidgetRef ref) {
  // Use for: navigation, showing SnackBar, playing sound
  ref.listen<AsyncValue<AuthState>>(authProvider, (prev, next) {
    if (next.hasError) {
      ScaffoldMessenger.of(context).showSnackBar(
        SnackBar(content: Text(next.error.toString()))
      );
    }
  });
  // ...
}
```

## Error Handling in Providers

```dart
// AsyncValue.guard — wraps future in try/catch, converts to AsyncError
state = await AsyncValue.guard(() async {
  return await someAsyncOperation();
});

// Manual error handling
Future<void> doSomething() async {
  state = const AsyncValue.loading();
  try {
    final result = await api.fetchData();
    state = AsyncValue.data(result);
  } on DioException catch (e) {
    state = AsyncValue.error(
      e.response?.data['error'] ?? 'Network error',
      StackTrace.current
    );
  }
}
```

## Testing with ProviderContainer

```dart
// test/habits_provider_test.dart
import 'package:flutter_riverpod/flutter_riverpod.dart';
import 'package:flutter_test/flutter_test.dart';

void main() {
  test('creates a habit', () async {
    // Create container with mock overrides
    final container = ProviderContainer(
      overrides: [
        habitsApiProvider.overrideWith((ref) => MockHabitsApi()),
      ],
    );
    addTearDown(container.dispose);

    // Start listening
    final subscription = container.listen(habitsProvider.future, (_, __) {});
    
    // Wait for initial load
    await container.read(habitsProvider.future);
    
    // Call notifier method
    await container.read(habitsProvider.notifier).createHabit('Exercise', 'daily');
    
    // Verify state
    final habits = await container.read(habitsProvider.future);
    expect(habits.any((h) => h.name == 'Exercise'), isTrue);
    
    subscription.close();
  });
}
```

## Anti-Patterns

```dart
// ❌ WRONG — ref.watch in callback
onPressed: () {
  final habits = ref.watch(habitsProvider);  // Watch only in build()
}

// ❌ WRONG — ref.read in build()
@override
Widget build(BuildContext context, WidgetRef ref) {
  final habits = ref.read(habitsProvider);  // Won't rebuild on changes
}

// ❌ WRONG — storing ref in class field (ref becomes stale)
class MyWidget extends ConsumerWidget {
  late WidgetRef _ref;
  
  @override
  Widget build(BuildContext context, WidgetRef ref) {
    _ref = ref;  // Don't do this
  }
}

// ❌ WRONG — accessing provider outside ProviderScope
// Every flutter_riverpod app needs ProviderScope at root
runApp(MyApp())  // Will crash if any widget uses ref

// ✅ CORRECT
runApp(const ProviderScope(child: MyApp()))
```
